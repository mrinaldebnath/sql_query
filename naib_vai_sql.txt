select * from concate
where concate not in
(select e1.concate as ee1
from concate e1 join concate e2
where abs(
TIMESTAMPDIFF
(SECOND,
STR_TO_DATE
(SUBSTRING_INDEX
(e1.concate,' ', -2),
'%d/%m/%Y %T'),
STR_TO_DATE
(SUBSTRING_INDEX
(e2.concate,' ', -2),
'%d/%m/%Y %T'))
)<=2
and
substring(e1.concate,1,length(e1.concate)-20)=
substring(e2.concate,1,length(e2.concate)-20)
and
e1.id!=e2.id
	)		
    order by id
	;
